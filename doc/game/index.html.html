    void webkitFullScreen();
    if (elem.requestFullscreen) {

      /* Safari */
      elem.webkitRequestFullscreen();
      elem.webkitEnterFullscreen(); // add
      document.webkitEnterFullscreen(); // add

      /* IE11 */
      elem.msRequestFullscreen();
